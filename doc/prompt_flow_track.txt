def checkStateSet(self):
    """当 QCheckBox 状态改变时触发（点击切换）"""
    super().checkStateSet()
    self.anim.stop()
    # 动画结束值：Dark 模式移动到右侧(36.0)，Light 模式回到左侧(4.0)
    self.anim.setEndValue(36.0 if self.isChecked() else 4.0)
    self.anim.start()
def set_theme_state(self, theme_name):
    """手动状态同步（初始化或主题强制应用时执行）"""
    self.theme_name = theme_name
    self.blockSignals(True)  # 避免触发点击回调，造成循环
    self.setChecked(theme_name == "Dark")
    self.blockSignals(False)
    self._thumb_pos = 36.0 if theme_name == "Dark" else 4.0
    self.update()
